# rtk-channel-selector-view

## Properties

| Property | Type | Required | Default | Description |
|----------|------|----------|---------|-------------|
| `channels` | `{
    id: string` | ✅ | - | Channels |
| `disableSearch` | `boolean` | ✅ | - | Disables search bar (default = false) |
| `hideAvatar` | `boolean` | ✅ | - | Hides avatar (default = false) |
| `iconPack` | `IconPack1` | ❌ | `defaultIconPack` | Icon Pack |
| `selectedChannelId` | `string` | ✅ | - | Selected channel id |
| `t` | `RtkI18n1` | ❌ | `useLanguage()` | Language |
| `viewAs` | `'dropdown' | 'list'` | ✅ | - | Render as dropdown or list (default = list) |

// app.module.ts
import { CUSTOM_ELEMENTS_SCHEMA } from '@angular/core';
import { defineCustomElements } from '@cloudflare/realtimekit-ui/loader';
